Cat\Env

.env 파일 파서. 환경변수 로드, 타입 캐스팅, 필수 키 검증

// .env 파일 로드
// APP_NAME=CatPHP
// APP_DEBUG=true
// DB_HOST=localhost

// 읽기 (기본값 지원)
$name  = env('APP_NAME', 'MyApp');    // 'CatPHP'
$debug = env('APP_DEBUG', false);     // true (자동 bool 변환)
$port  = env('DB_PORT', 3306);        // 3306 (미설정 → 기본값)

// 필수 키 검증
env()->required(['APP_KEY', 'DB_HOST']);

Cat\Request

HTTP 요청 추상화. input, query, header, file, bearerToken, method, ajax 감지

// 입력 읽기
$name  = request()->input('name');           // POST/GET 자동 감지
$page  = request()->query('page', 1);        // GET 파라미터
$all   = request()->all();                   // 전체 입력

// 헤더 + 인증
$token = request()->bearerToken();          // Authorization: Bearer xxx
$type  = request()->header('Content-Type');

// 메서드 + 감지
request()->method();     // 'GET'
request()->isAjax();     // X-Requested-With 확인
request()->isSecure();   // HTTPS 여부

output Response
Cat\Response

HTTP 응답 빌더. redirect, download, stream, cache, noContent, 커스텀 헤더

// 리다이렉트
response()->redirect('/login');
response()->redirect('/new-url', 301);  // 영구 이동

// 파일 다운로드
response()->download('storage/report.pdf', '리포트.pdf');

// 캐시 헤더
response()->cache(3600);     // Cache-Control: max-age=3600
response()->noCache();         // no-store, no-cache

// 상태 코드
response()->noContent();       // 204 No Content
response()->status(403);       // 403 Forbidden

// 커스텀 헤더
response()->header('X-Custom', 'value')->send();
응답 타입 비교
메서드상태Content-Type용도

stream()200자동 감지인라인 파일 표시
noContent()204—빈 응답 (DELETE 등)

Cat\Session

세션 관리. get/set, flash (1회성), regenerate (세션 고정 방지), destroy

// 읽기 / 쓰기
session('user_id');                 // 읽기 (shortcut)
session()->set('user_id', 1);        // 쓰기
session()->get('user_id', 0);        // 기본값 포함 읽기

// Flash (1회성 — 다음 요청에서 자동 삭제)
session()->flash('success', '저장 완료!');
$msg = session()->getFlash('success');

// 보안
session()->regenerate();  // 세션 ID 재생성 (고정 방지)
session()->destroy();     // 완전 삭제 (로그아웃)

Cat\Migration

DB 스키마 버전 관리. up/down 마이그레이션, 롤백, 상태 확인

// 마이그레이션 파일 예시
// migrations/2024_01_15_create_users_table.php
return [
    'up' => fn() => db()->exec('
        CREATE TABLE users (
            id INT AUTO_INCREMENT PRIMARY KEY,
            name VARCHAR(100) NOT NULL,
            email VARCHAR(255) UNIQUE NOT NULL,
            password VARCHAR(255) NOT NULL,
            role ENUM("admin","user") DEFAULT "user",
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    '),
    'down' => fn() => db()->exec('DROP TABLE IF EXISTS users'),
];

이미지/수학 캡차 생성 (GD 라이브러리). 세션 기반 검증, 커스텀 난이도

// 이미지 캡차
$src = captcha()->src();   // data:image/png;base64,...
echo "<img src='{$src}'>";

// 수학 캡차
$q = captcha()->math();  // ['question' => '3 + 7 = ?', 'answer' => 10]

// 검증
if (!captcha()->verify($_POST['captcha'])) {
    json()->fail('캡차 인증 실패');
}

person User
Cat\User

유저 CRUD + 인증 통합. 가입, 로그인, 프로필 수정, 비밀번호 변경

// 회원가입
$id = user()->create([
    'name'     => '김개발',
    'email'    => 'kim@catphp.dev',
    'password' => 'secret123',  // 자동 Argon2id 해싱
]);

// 로그인 (email + password 검증 → 세션 시작)
$ok = user()->attempt('kim@catphp.dev', 'secret123');

// 현재 유저
$me = user()->current();

// 프로필 수정
user()->update($id, ['name' => '김CatPHP']);

// 비밀번호 변경 (update에 password 포함 시 자동 Argon2id 해싱)
user()->update($id, ['password' => 'newPass']);

// 삭제
user()->delete($id);

// ID로 조회 / 필드로 조회 (XSS 자동 살균)
$user = user()->find(1);
$user = user()->findBy('email', 'kim@catphp.dev');
