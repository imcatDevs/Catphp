Cat\Mail

순수 PHP 소켓 SMTP 클라이언트. 외부 의존성 없이 메일 전송, 첨부파일, 템플릿

// 기본 전송
mailer()->to('user@example.com')
    ->subject('가입을 환영합니다!')
    ->body('<h1>환영합니다</h1><p>CatPHP에 가입해주셔서 감사합니다.</p>')
    ->send();

// CC + BCC + 첨부파일
mailer()->to('to@example.com')
    ->cc('cc@example.com')
    ->bcc('bcc@example.com')
    ->subject('월간 리포트')
    ->template('emails/report', ['month' => '1월'])
    ->attach('/path/to/report.pdf')
    ->send();

// 미리보기 (전송 없이 HTML 확인)
$html = mailer()->to('test@example.com')
    ->subject('테스트')->body('Hello')->preview();

queue Queue
Cat\Queue

비동기 작업 큐. Redis/DB 드라이버, 지연 실행, 재시도, 실패 관리

// 핸들러 등록
queue()->handle('send-email', function(array $payload) {
    mailer()->to($payload['to'])
        ->subject($payload['subject'])
        ->body($payload['body'])->send();
});

// 작업 추가
queue()->push('send-email', [
    'to' => 'user@example.com',
    'subject' => '환영합니다',
    'body' => '...',
]);

// 5분 후 실행
queue()->later(300, 'send-email', $payload);

// 워커 실행
queue()->work(null, 3, 100);  // sleep 3초, 최대 100개

schedule Schedule
Cat\Schedule

Cron 스케줄러. PHP 코드로 주기적 태스크 등록, 시스템 crontab 한 줄만 설정

// 콜백 태스크 등록 → 스케줄 체이닝
schedule()->call(fn() => cache()->clear())->daily();
schedule()->call(fn() => logger()->info('Heartbeat'))->hourly();
schedule()->call(fn() => queue()->pop())->everyMinute();

// CLI 명령어 + cron 표현식
schedule()->command('log:rotate')->weeklyOn(1, '03:00'); // 매주 월요일 03:00

// 시스템 crontab 설정 (1줄)
// * * * * * php /path/to/cli.php schedule:run >> /dev/null 2>&1

fingerprint Hash
Cat\Hash

파일/문자열 해싱, HMAC 서명, 체크섬 무결성 검증

// 문자열 해시
$sha256 = hasher()->string('Hello');            // SHA-256
$md5    = hasher()->string('Hello', 'md5');    // MD5

// HMAC 서명
$hmac = hasher()->hmac('payload', 'secret-key');
hasher()->verifyHmac('payload', $hmac, 'secret-key');

// 파일 체크섬
$checksum = hasher()->file('/path/to/file.zip');
hasher()->verify('/path/to/file.zip', $checksum);
